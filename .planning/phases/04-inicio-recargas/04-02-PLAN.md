    ```ts
      empresa_id: string;

    }

    }
    ```

    Export ONE function:

    ```ts
    /**
     * Find empresas whose FIRST-EVER transaction in the dataset falls
     * within the active filter window. Caps shown to top 5 (sorted by
     * firstTx ascending — earliest activation first); overflowCount
     * carries the rest for "+N más" display.
     *
     * IMPORTANT (Pitfall 5 in 04-RESEARCH.md): callers MUST pass the
     * FULL transaction dataset (allTx from getCachedTransactions), NOT
     * the period-filtered subset. If you pass a filtered subset, every
     * empresa in that subset will appear "new" because their earliest
     * tx in the filtered subset is by definition in-period.
     *
     * Cost: O(N) single pass over allTx (~3188 rows in production), well
     * under 5ms. No second fetch needed — getCachedTransactions is React
     * cache()-deduped per request.
     *
     * Returns { shown: [], overflowCount: 0 } when no empresas are new.
     * Returns { shown: [], overflowCount: 0 } when filters.from/to are
     * missing (no defined window).
     */
    export function findEmpresasNuevasActivadas(
      allTransactions: Transaction[],
      filters: DashboardFilters,
    ): EmpresasNuevasResult;
    ```

    Implementation:
    - Window guard: `if (!filters.from || !filters.to) return { shown: [], overflowCount: 0 };`
    - Construct window bounds as numeric timestamps:
      ```ts
      const windowStart = new Date(`${filters.from}T00:00:00-05:00`).getTime();
      const windowEnd = new Date(`${filters.to}T23:59:59.999-05:00`).getTime();
      ```
    - Build `firstTxByEmpresa: Map<empresa_id, { fecha: Date; nombre: string }>` in single pass:
      ```ts
      for (const tx of allTransactions) {
        const cur = firstTxByEmpresa.get(tx.empresa_id);
        if (!cur || tx.fecha.getTime() < cur.fecha.getTime()) {
          firstTxByEmpresa.set(tx.empresa_id, { fecha: tx.fecha, nombre: tx.empresa_nombre });
        }
      }
      ```
    - Filter map entries to `firstTx ∈ [windowStart, windowEnd]`.
    - Sort ascending by `firstTx`.
    - If empresa filter is active (`filters.empresa` is set and not the `__all__` sentinel), filter results to that single empresa BEFORE the cap (CONTEXT.md cliente-foco contract: hechos curados hide entirely in cliente-foco via CSS; this filter ensures sane shape if ever rendered).
    - `shown = sortedResults.slice(0, 5)`, `overflowCount = Math.max(0, sortedResults.length - 5)`.
    - Return `{ shown, overflowCount }`.

    Add inline doc-comment explaining why caching the result is unnecessary (already O(N) under 5ms; React cache dedupe handles upstream fetch).
