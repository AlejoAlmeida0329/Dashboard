   ```ts
   import { toZonedTime, formatInTimeZone } from 'date-fns-tz';
   import { es as esLocale } from 'date-fns/locale';

   export const BOGOTA_TZ = 'America/Bogota';

   const cop = new Intl.NumberFormat('es-CO', {
     style: 'currency',
     currency: 'COP',
     maximumFractionDigits: 0, // COP sin centavos en práctica
   });

   export const formatCOP = (n: number | null | undefined): string => {
     if (n === null || n === undefined || !Number.isFinite(n)) return '—';
     return cop.format(n);
   };

   const intCO = new Intl.NumberFormat('es-CO', { maximumFractionDigits: 0 });
   export const formatInteger = (n: number | null | undefined): string => {
     if (n === null || n === undefined || !Number.isFinite(n)) return '—';
     return intCO.format(n);
   };

   const pctCO = new Intl.NumberFormat('es-CO', { style: 'percent', maximumFractionDigits: 1 });
   export const formatPercent = (n: number | null | undefined): string => {
     if (n === null || n === undefined || !Number.isFinite(n)) return '—';
     return pctCO.format(n);
   };

   export const formatBogotaDate = (d: Date): string =>
     formatInTimeZone(d, BOGOTA_TZ, 'dd/MM/yyyy', { locale: esLocale });

   export const formatBogotaDateTime = (d: Date): string =>
     formatInTimeZone(d, BOGOTA_TZ, "dd/MM/yyyy HH:mm:ss 'COT'", { locale: esLocale });

   export const todayISOInBogota = (): string =>
     formatInTimeZone(new Date(), BOGOTA_TZ, 'yyyy-MM-dd');
   ```

2. **`src/lib/url-state.ts`** — helpers para preservar searchParams al navegar:
   ```ts
   export type DashboardFilters = {
     from?: string;       // 'YYYY-MM-DD'
     to?: string;
     empresa?: string;    // empresa_id
     presenter?: '1';
   };

   export function parseFilters(searchParams: Record<string, string | string[] | undefined>): DashboardFilters {
     const get = (k: string) => {
       const v = searchParams[k];
       return Array.isArray(v) ? v[0] : v;
     };
     return {
       from: get('from'),
       to: get('to'),
       empresa: get('empresa'),
       presenter: get('presenter') === '1' ? '1' : undefined,
     };
   }

   export function buildUrl(pathname: string, filters: DashboardFilters): string {
     const params = new URLSearchParams();
     if (filters.from) params.set('from', filters.from);
     if (filters.to) params.set('to', filters.to);
     if (filters.empresa) params.set('empresa', filters.empresa);
     if (filters.presenter === '1') params.set('presenter', '1');
     const qs = params.toString();
     return qs ? `${pathname}?${qs}` : pathname;
   }

   export function presetDateRange(preset: '7d' | '30d' | 'mtd' | 'qtd' | 'ytd'): { from: string; to: string } {

   }
   ```
