   // El adapter pasa un objeto {header_lowercased: cellValue} construido con headerIndexMap
   export const TransactionRowSchema = z.object({
     // Nombres de header tentativos — alinear con el Sheet real cuando se confirme:
     fecha: z.coerce.date(),

     monto: parsed.monto,
     tipo: parsed.tipo as TransactionType,

1. **`src/lib/sheets/transactions.ts`** (header `'server-only'`):
   ```ts
   import 'server-only';
   import { getSheetsClient } from './client';
   import { SPREADSHEETS } from './config';
   import { headerIndexMap, isEmptyRow, isFormulaError, withRetry } from './_utils';
   import { TransactionRowSchema, ExpectedTransactionHeaders } from '@/lib/domain/schemas';
   import type { Transaction } from '@/lib/domain/types';

   export interface AdapterResult<T> {
     rows: T[];
     skipped: number;
     lastReadAt: Date;
     warnings: string[];
   }

   export async function getTransactions(): Promise<AdapterResult<Transaction>> {
     const sheets = getSheetsClient();
     const lastReadAt = new Date();
     const res = await withRetry(() => sheets.spreadsheets.values.get({
       spreadsheetId: SPREADSHEETS.transactions.id,
       range: SPREADSHEETS.transactions.range,
       valueRenderOption: 'UNFORMATTED_VALUE',
       dateTimeRenderOption: 'FORMATTED_STRING',
     }));

     const allRows = res.data.values ?? [];
     if (allRows.length < 2) {
       return { rows: [], skipped: 0, lastReadAt, warnings: ['Sheet vacío o sin headers'] };
     }
     const map = headerIndexMap(allRows[0]);

     // Boot-time schema check — fail loud, not silent (Pitfall 3):
     const missingHeaders = ExpectedTransactionHeaders.filter(h => !map.has(h));
     if (missingHeaders.length > 0) {
       throw new Error(
         `Sheet schema mismatch — columnas faltantes en transactions Sheet: ${missingHeaders.join(', ')}. ` +
         `Verifica el Sheet o ajusta src/lib/domain/schemas.ts si los nombres cambiaron upstream.`
       );
     }

     const rows: Transaction[] = [];
     let skipped = 0;
     const warnings: string[] = [];

     for (const raw of allRows.slice(1)) {
       if (isEmptyRow(raw)) { skipped++; continue; }
       const obj: Record<string, unknown> = {};
       for (const [header, idx] of map.entries()) {
         const cell = raw[idx];
         if (isFormulaError(cell)) { skipped++; warnings.push(`Formula error en ${header}: ${cell}`); continue; }
         obj[header] = cell;
       }
       const parsed = TransactionRowSchema.safeParse(obj);
       if (!parsed.success) { skipped++; continue; }
       rows.push(parsed.data);
     }

     return { rows, skipped, lastReadAt, warnings };
   }
   ```

2. **`src/app/api/_smoke/route.ts`** — Route Handler que ejercita el adapter:
   ```ts
   import { NextResponse } from 'next/server';
   import { verifySession } from '@/lib/auth/dal';
   import { getTransactions } from '@/lib/sheets/transactions';

   export const dynamic = 'force-dynamic';

   export async function GET() {
     await verifySession(); // protegido por auth
     try {
       const result = await getTransactions();
       return NextResponse.json({
         ok: true,
         count: result.rows.length,
         skipped: result.skipped,
         lastReadAt: result.lastReadAt.toISOString(),
         warnings: result.warnings.slice(0, 5),
         sample: result.rows.slice(0, 3),
       });
     } catch (err) {
       return NextResponse.json({
         ok: false,
         error: err instanceof Error ? err.message : 'unknown error',
       }, { status: 500 });
     }
   }
   ```

**Importante**: El smoke endpoint está protegido por auth (verifySession). Para curl-testing en Plan 04, hay que tener cookie session válida (login primero).
