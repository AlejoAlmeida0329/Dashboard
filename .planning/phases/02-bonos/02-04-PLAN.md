**Estructura del file:**

```tsx
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";

import { BonosChart } from "@/components/bonos/BonosChart";
import { KPICards } from "@/components/bonos/KPICards";
import { Leaderboard } from "@/components/bonos/Leaderboard";
import { SalesTable } from "@/components/bonos/SalesTable";

import {
  aggregateBonosByDate,
  aggregateBonosByEmpresa,
  filterBonos,
  summarizeBonos,
  top10Empresas,
} from "@/lib/domain/bonos";
import { getCachedTransactions } from "@/lib/sheets/transactions";
import { parseFilters } from "@/lib/url-state";

export const metadata = {
  title: "Bonos · Tikin Dashboard",
};

// Force dynamic — la página depende de URL state y de un fetch externo
// que NO queremos cachear cross-request (la fuente es Sheets, debe leerse fresh).
export const dynamic = "force-dynamic";

type PageProps = {
  // Next 16: searchParams es una Promise en la firma del Server Component.
  searchParams: Promise<Record<string, string | string[] | undefined>>;
};

export default async function BonosPage({ searchParams }: PageProps) {
  const params = await searchParams;
  const filters = parseFilters(params);

  let result;
  try {
    result = await getCachedTransactions();
  } catch (err) {
    return (
      <Card>
        <CardHeader>
          <CardTitle>No pudimos leer el Sheet</CardTitle>
        </CardHeader>
        <CardContent>
          <p className="text-sm text-muted-foreground">
            {err instanceof Error ? err.message : "Error desconocido al leer transacciones."}
          </p>
        </CardContent>
      </Card>
    );
  }

  const bonos = filterBonos(result.rows, filters);
  const summary = summarizeBonos(bonos);
  const byDate = aggregateBonosByDate(bonos);
  const byEmpresa = aggregateBonosByEmpresa(bonos);
  const top10 = top10Empresas(byEmpresa);

  if (bonos.length === 0) {
    return (
      <div className="space-y-6">
        <KPICards summary={summary} />
        <Card>
          <CardHeader>
            <CardTitle>Sin bonos en el período seleccionado</CardTitle>
          </CardHeader>
          <CardContent>
            <p className="text-sm text-muted-foreground">
              Probá ampliando el rango de fechas o quitando el filtro de empresa.
            </p>
          </CardContent>
        </Card>
      </div>
    );
  }

  return (
    <div className="space-y-6">
      <KPICards summary={summary} />

      <Card>
        <CardHeader>
          <CardTitle>Bonos vendidos en el tiempo</CardTitle>
        </CardHeader>
        <CardContent>
          <BonosChart data={byDate} />
        </CardContent>
      </Card>

      <div className="grid gap-6 lg:grid-cols-[1fr_2fr]">
        <Leaderboard rows={top10} />
        <SalesTable rows={byEmpresa} />
      </div>
    </div>
  );
}
```

**Detalles de implementación:**

1. **`searchParams: Promise<...>`** — Next 16 cambió la firma del page Server Component a Promise. `await searchParams` es la forma correcta. Si el ejecutor encuentra que la versión instalada usa la firma sincrónica (sin Promise), ajustar. La detección rápida: `cat node_modules/next/package.json | jq -r .version`.

2. **`dynamic = "force-dynamic"`** — la página depende de URL state (no cachear) y debe re-fetchar al Sheet en cada visita.

3. **Error fallback inline** — si `getCachedTransactions()` falla, mostrar un Card con el mensaje del error. NO usar `error.tsx` global porque ese se reservó para errores de runtime (ya existe en el route group). El catch local nos permite copy más específico ("No pudimos leer el Sheet").

4. **Empty state** — cuando `bonos.length === 0` (filtro produce 0 resultados, no error), mostrar KPICards (con summary cero) + Card con copy amigable. KPICards funciona con summary cero porque Plan 02 garantizó `count=0 → ticketPromedio=0` (no NaN).

5. **Layout responsive:**
   - Mobile (default): todo stacked en columna, espaciado `space-y-6`.
   - Desktop (`lg:`): Leaderboard | SalesTable lado a lado, ratio 1:2 (la tabla pesa más por las múltiples columnas). Chart hero ocupa width completo.
   - KPICards ya es internamente responsive (sm:grid-cols-2).

6. **Modo Presentación + filtro empresa = "vista cliente foco":**
   - Cuando `filters.empresa` está set, `aggregateBonosByEmpresa` retornará solo 1 row (la empresa filtrada). `top10Empresas` retornará esa misma 1 row (length < 10 está soportado).
   - Cuando además `filters.presenter === '1'`, el CSS de Phase 1 oculta `data-presenter-hide` elementos: el Leaderboard entero (que ahora tiene 1 entrada irrelevante), el KPI Comisión, y las cols sensibles de la tabla.
   - El resultado: la página se transforma en "esta empresa, sus bonos, su tendencia" — exactamente lo que 02-CONTEXT.md describe como momento clave.

7. **Sin imports inutiles** — no traer `verifySession()` aquí (el `ProtectedLayout` ya lo hace). No re-fetchar empresas (DashboardHeader las tiene).

8. **i18n consistente** — todo el copy en español, mismos verbos del resto de la app ("Probá", "Sin bonos en el período seleccionado", etc.).
