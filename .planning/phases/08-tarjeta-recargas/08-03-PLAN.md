    ```ts
      upperCOP: number | null;      // null for the top bucket
      count: number;
      volumenCOP: number;
    }

    export interface TopRecharger {
      tikintag: string;
      empresa: string | undefined;  // best-effort label from first row's transferEmpresa
      recargas: number;
      volumenCOP: number;
      recargaPromedio: number;
    }
    ```

    Implement `aggregateRechargeAdoption(allTx: Transaction[], recargaRows: Transaction[]): RechargeAdoption`:
    - Same shape as `aggregatePurchaseAdoption` in `cardUsage.ts` (Plan 08-01) — distinct
      tikintags numerator + denominator, ratio.
    - Cite REC-V2-03 + PRD baseline 40/235 ≈ 17% adoption.

    Implement `aggregateRechargeMethodSplit(rows: Transaction[]): RechargeMethodSplit`:
    - Partition by `tx.tipo` into PSE vs TRANSFER buckets.
    - Per bucket: count, sum Math.abs(amount), share = count / totalCount.
    - When totalCount === 0: both shares = 0 (zero-safe).
    - Return literal shape, NOT array — page binds two cards directly off `result.pse` /
      `result.transfer`.

    Implement `aggregateRechargeAmountDistribution(rows: Transaction[]): RechargeAmountBucket[]`:
    - Three fixed buckets in this order (matches REC-V2-06 spec):
      1. `{ label: "<$100K", lowerCOP: null, upperCOP: 100_000, ... }` — `amount < 100k`
      2. `{ label: "$100K-$1M", lowerCOP: 100_000, upperCOP: 1_000_000, ... }` — `100k <= amount <= 1M`
      3. `{ label: ">$1M", lowerCOP: 1_000_000, upperCOP: null, ... }` — `amount > 1M`
    - For each row, classify via `Math.abs(tx.transactionAmount)`.
    - Always return all 3 buckets even when some are zero (chart needs stable axis).
    - Document boundary inclusivity unambiguously in JSDoc (`<` vs `<=`).

    Implement `aggregateTopRechargers(rows: Transaction[], limit: number = 10): TopRecharger[]`:
    - Group by `tx.transferTikintag` (skip undefined defensively).
    - Per group: count, sum Math.abs(amount), capture first-row `transferEmpresa` as label.
    - Sort by `volumenCOP` DESC; slice to `limit` (default 10 per REC-V2-07).

    Same conventions as v2 functions in `bonos.ts` / `payouts.ts` — no defensive any-typing,
    no NaN paths, JSDoc cites the matching REQ-ID for traceability.
